Operator

Operator is a kind of token.

#include <litan.h>

int main() {

    // arithmetic operator
    1 + 1; // 2
    3 * 4; // 12
    14 / 3; // 4
    11 % 3; // 2

    // comparative operator
    5 < 6; // true
    5 >= 6; // false
    5 == 6; // false
    5 != 6; // true

    // logic operator
    !true; // false
    !false; // true
    true && false; // false
    true || false; // true

    return ;
}

Operant are literals or variable

A unary operator is used in syntax like <operator> <operant> or <operant> <operator>.

+3 !true &some_variable

A binary operator is used as <left operant> <operator> <right operant>.

2 + 3 5 != 6 true || false pi = 3.14

Playing with Array and Pointer using * & []

#include <litan.h>

int main() {

    // Step 0 - Initialization

    // declare an array to int with size 3, named `a`
    int a[3];
    // `a[i]` is the `i`th element in the array `a`
    // assign value to each element of the array
    a[0] = 1997;
    a[1] = 11;
    a[2] = 11;

    // declare an pointer to int, named `p`
    int *p;
    // assign the address of the first element in the array 
    p = &a[0];


    // Step 1 - Playing Array and Pointer Separately

    // playing with array
    // get value from array by `[]` operator
    a[0]; // 1997
    a[1]; // 11
    a[2]; // 11
    a[3]; // undefined, index out of bound
    // set value of birthday to lunar 
    a[1] = 10;
    a[2] = 12;


    // checking `p` is equal to `&a[0]`
    p == &a[0]; // true
    (p + 1) == &a[1]; // true
    (p + 2) == &a[2]; // true
    *p == a[0]; // true
    *(p + 1) == a[1]; // true
    *(p + 2) == a[2]; // true


    // playing with pointer
    *p; // 1997
    *(p + 1); // 10
    *(p + 2); // 12
    // set value of birthday back to geogorian
    *(p + 1) = 11;
    *(p + 2) = 11;


    // Step 2 - Playing Array and Pointer with Each Other

    // checking `p` is equal to `a`
    p == a; // true
    p + 1 == a + 1; // true
    p + 2 == a + 2; // true

    // playing array as an pointer
    *(a + 1); // 11
    *(a + 2); // 11
    *(a + 1) = 10;
    *(a + 2) = 12;

    // playing pointer as an array
    // `a[i]` is equivalent to `*(a + i)`
    p[0]; // 1997
    p[1]; // 11
    p[2]; // 11
    p[1] = 10;
    p[2] = 12;


    // Step 3 - Modifing Pointer
    // let p points to another place
    p = &a[1];
    *p; // 10
    *(p - 1); // 1997
    *(p + 1); // 12
    *(p + 2); // error

    p = p + 1;
    *(p - 2); // 1997
    *(p - 1); // 10
    *p; // 12
    *(p + 1); // error

    // set p back to ;
    p = a;


    // Step 4 - Playing Array and Pointer in A Great Messiness

    // playing integer as an pointer or pointer
    a[i] == *(a + i); // true
    i[a] == *(i + a); // true
    *(a + i) == *(i + a); // true

    // don't code like this
    1[p]; // 11
    2[p]; // 11
    1[p] = 10;
    2[p] = 12;
    1[a]; // 10
    2[a]; // 12
    1[a] = 10;
    2[a] = 12;

    a[0] == 0[a]; // true
    p[0] == 0[p]; // true

    &a[0] == a; // true
    &*p == p; // true
    &*&*p == p; // true


    // Step 5 - Pointer Points not only to array;
    int x = 3;
    p = &x;
    *p; // 3
    *p = 5;
    x; // 5


    // it is time to stop playing
    pause();

    return 0;
}
