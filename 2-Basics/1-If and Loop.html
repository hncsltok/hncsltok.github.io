If Statement

The statements inside {} are executed if the expression inside () is true, and will be skipped if the expression is false.

if (<expression>) { <statements> }


#include <litan.h>

int main() {

    put_string("input an interger: ");
    int x = get_int();

    if(x < 0) {
        x = -x;
    }

    put_string("The absolute value of it is : ");
    put_int(x);

    pause();

    return 0;
}

All non-zero value is regarded as true while zero value is regarded as false.

Zero Value Non-Zero Value
false 0 '\0' "" true 1 'a' " " 

#include <litan.h>

int main() {

    if(1) {
        put_string("`1` is regarded as true!\n");
    }

    return 0;
}

If-Else Statement

if (<expression>) { <statements> } else { <statements> }

#include <litan.h>

int main() {

    put_string("Welcome to Litan\'s Secret Database.\n");
    put_string("Password:");
    int password = get_int();

    if(password == 2333) {
        put_string("Access Granted");
    }
    else {
        put_string("Access Denied");
    }

    pause();

    return 0;
}

While Statement

while (<expression>) { <statements> }

#include <litan.h>

int main() {

    put_string("Input numbers to sum them up.");
    warp();

    int inputted = -1;
    int sum = 0;

    while(inputted) {
        inputted = get_int();
        sum += inputted;

        put_string("Now the sum is ");
        put_int(sum);
        warp();
    }

    return 0;
}
