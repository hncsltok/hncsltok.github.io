Array

<type> <name>[<length>]; is an array definition statement.

By int a[100]; array definition statement, an array variable is defined, named a. It contains 100 numbers int.

<length> must be an integer, and it must be constant or constant expression. This is because the size of memory it has must be fixed.

#include <litan.h>

int main() {

    const int width = 3;
    const int height = 4;
    int a[width * height]; // legal, width and height are both constant

    int x = get_int();
    int b[x]; // error, x is not constant before running.

    return 0;
}
#include <litan.h>

int main() {

    const int length = 100;
    int a[length];

    int i;
    for (i = 1; i <= length; ++i) {
        a[i] = i;        
    }

    int sum;
    for (i = 1; i <= length; ++i) {
        sum += a[i];
    }

    put_string("the sum from 1 to 100 is :");
    put_int(sum);

    pause();

    return 0;
}

Array Initialization Statement

<type> <name>[<length>] = { <expression>, <expression>, ...};

#include <litan.h>

int main() {

    // defining an array and initialize it separately
    int birthday_geogorian[3];
    birthday_geogorian[0] = 1997;
    birthday_geogorian[1] = 11;
    birthday_geogorian[2] = 11;

    // defining an array and initialize with `{}`
    int birthday_lunar[3]  = {1997, 10, 12};

    pause();

    return 0;
}

Pointer

#include <litan.h>

int main() {

    const int length = 100;
    int a[length];

    int i;
    for (i = 1; i <= length; ++i) {
        a[i] = i;        
    }

    int *p;
    p = a;

    put_string("The first element in `a` is");
    put_int(*p);

    ++p;
    put_string("The next element in `a` is")
    put_int(*p);

    p = a + 10;
    put_string("The 10th element after a[0] is");
    put_int(*p);

    put_string("The 10th element after current *p is");
    put_int(*(p + 10));

    pause();

    return 0;
}
